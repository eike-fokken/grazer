51  double euler_maruyama_oup(
52  double stability_parameter, double cut_off_factor, double last_value,
53  double theta, double mu, double delta_t, double sigma,
54  Distribution &distribution, int number_of_stochastic_steps) {
55  number_of_stochastic_steps = compute_stable_number_of_oup_steps(
56  stability_parameter, theta, delta_t, number_of_stochastic_steps);
57  double stochastic_stepsize = delta_t / number_of_stochastic_steps;
58  double current_value = last_value;
59  for (auto i = 0; i != number_of_stochastic_steps; ++i) {
60  current_value
61  = current_value + theta * (mu - current_value) * stochastic_stepsize
62  + sigma * distribution.get_sample(sqrt(stochastic_stepsize));
63  if (sgn(mu) * current_value < (1.0 - cut_off_factor) * sgn(mu) * mu) {
64  current_value = (1.0 - cut_off_factor) * mu;
65  } else if (
66  sgn(mu) * current_value > (1.0 + cut_off_factor) * sgn(mu) * mu) {
67  current_value = (1.0 + cut_off_factor) * mu;
68  }
69  }
70  return current_value;
71  }
