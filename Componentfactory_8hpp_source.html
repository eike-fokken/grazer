63  nlohmann::json get_component_boundary_schema(
64  nlohmann::json const &defaults, bool allow_required_defaults) {
65  static_assert(
66  std::is_base_of_v<Boundaryvaluecomponent, Component>,
67  "This function can only be called with a class deriving from "

300  std::unique_ptr<Network::Node>
301  make_instance(nlohmann::json &topology) const final {
302  nlohmann::json topology_schema = get_schema(
303  /*allow_required_defaults=*/true, /*include_external=*/true);
304  Aux::schema::apply_defaults(topology, topology_schema);
305  // order is important (defaults can be required!)
306  Aux::schema::validate_json(topology, topology_schema);
307 
308  check_class_hierarchy_properties<ConcreteNode>();
309  return std::make_unique<ConcreteNode>(topology);
